常用命令积累

2018-08-13

这篇文章主要记录自己在学习中遇到的一些常用的命令

Windows

tasklist　　　　　　 //查看当前运行的进程
net user　　　　 //查看用户
systeminfo　　　　 //查看电脑详细信息
del + 文件名　　　 　//删除文件
lusrmgr.msc　　 //运行框打开用户配置
services.msc　 　 //运行框查看服务
shutdown -s -f -t 0　　　　 //关机
ipconfig | find "IPv4"　　　　 //过滤参数
taskkill /f /im firefox.exe　　　　//关闭电脑某进程
copy 1.txt/b + 1.png/b 2.png　　//合并1.txt和1.png为2.png
rundll32.exe user32.dll,LockWorkStation　　//锁屏
netsh wlan show profiles name="TaoTao" key=clear　　//查看wifi密码
for /L %I in (1,1,254) DO @ping -w 1 -n 1 192.168.200.%I | findstr "TTL=" >> result.txt //利用ICMP扫描200网段的所有机器

Linux

free -m　　//查看内存使用情况
df -m　　//查看储存使用情况
chmod +x *　　//给予当前目录所有文件执行权限
netstat -nao　　//查看打开的端口
taskkill /f -pid 1220　　//关闭占用端口的进程

Web

alert(document.lastModified);　　//判断是否为伪静态

Mysql

show variables like 'char%';　　//查看数据库编码

Kali

leafpad /etc/apt/sources.list　　//更换软件源
sudo apt-get update && apt-get upgrade -y　　//更新软件包和依赖
set exitonsession false　　//vps上设置可以回连多个回话
set enablestageencoding true　　//双端加密
set stagerverifysslcert false　　//ssl
set stageencodingfallback false　　//返回加密
idletime　　//meterpreter查看主机空闲时间
apt-get clean & apt-get autoremove　　//清理下载的空间
